<script>
// --- CAMBIO DE SECCIONES PRINCIPALES ---
const menuBtns = document.querySelectorAll('.menu-btn');
const sections = document.querySelectorAll('.section');

menuBtns.forEach(btn => {
  btn.addEventListener('click', e => {
    e.preventDefault();
    menuBtns.forEach(b => b.classList.remove('active'));
    sections.forEach(s => s.classList.remove('visible'));
    btn.classList.add('active');
    document.getElementById(btn.dataset.section).classList.add('visible');
  });
});

// --- BOTONES DE TORNEOS EN CAMPEONES ---
const torneoBtns = document.querySelectorAll('.torneo-btn');
const torneoLists = document.querySelectorAll('.torneo-list');

torneoBtns.forEach(btn => {
  btn.addEventListener('click', () => {
    torneoBtns.forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    torneoLists.forEach(list => list.classList.remove('visible'));
    const target = document.getElementById(btn.dataset.torneo);
    target.classList.add('visible');
  });
});
</script>
